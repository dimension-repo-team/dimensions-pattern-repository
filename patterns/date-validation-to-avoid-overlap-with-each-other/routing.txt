Function Val_QSSequencePrevious(Loop_QSSequencePrevious, IOM, Attempt)
	
	Dim cat, catx, cat1, cat2, CurrentMonth, MRK_QSBIOLOGICDURATIONx, MRK_QSSequencePrevious, MRK_QSSP_X
		
	set CurrentMonth = IOM.Questions["CurrentMonth"]
	set MRK_QSBIOLOGICDURATIONx = IOM.Questions["MRK_QSBIOLOGICDURATIONx"]
	set MRK_QSSequencePrevious = IOM.Questions["MRK_QSSequencePrevious"]
	set MRK_QSSP_X = IOM.Questions["MRK_QSSP_X"]
	
	For each cat in Loop_QSSequencePrevious.Categories
		For each catx in Loop_QSSequencePrevious[cat].Grid_QSSP.Categories
			dim MonthActual
			MonthActual = 0
			
			if Loop_QSSequencePrevious[cat].Grid_QSSP[catx].QSSP_2 = 2025 then
				MonthActual = Loop_QSSequencePrevious[cat].Grid_QSSP[catx].QSSP_1.Categories[Loop_QSSequencePrevious[cat].Grid_QSSP[catx].QSSP_1].Properties["month"]		
				'debug.MsgBox(MonthActual)				
				if MonthActual > CurrentMonth then
			    	Loop_QSSequencePrevious.Errors.Clear()
			    	Loop_QSSequencePrevious.Errors.AddNew("ERR31",Makestring(catx.Label,": ",IOM.Questions["hCustomMsgs"].Categories.ERR31.Label))
			    	Val_QSSequencePrevious = false
			    	exit function
			    end if	
			end if
			
			dim MonthDiff_Start, MonthDiff_Stop, YearDiff_Start, YearDiff_Stop, ActualMonth_Start, ActualMonth_Stop
			MonthDiff_Start = 0
			MonthDiff_Stop = 0
			YearDiff_Start = 0
			YearDiff_Stop = 0
			ActualMonth_Start = 0
			ActualMonth_Stop = 0
			
			MonthDiff_Start = month(now()) - Loop_QSSequencePrevious[{Started}].Grid_QSSP[catx].QSSP_1.Categories[Loop_QSSequencePrevious[{Started}].Grid_QSSP[catx].QSSP_1].Properties["month"]
			YearDiff_Start = year(now()) - Loop_QSSequencePrevious[{Started}].Grid_QSSP[catx].QSSP_2
		    ActualMonth_Start = (YearDiff_Start * 12) + MonthDiff_Start 
			'debug.MsgBox(ActualMonth_Start)
			
			MonthDiff_Stop = month(now()) - Loop_QSSequencePrevious[{Stopped}].Grid_QSSP[catx].QSSP_1.Categories[Loop_QSSequencePrevious[{Stopped}].Grid_QSSP[catx].QSSP_1].Properties["month"]
			YearDiff_Stop = year(now()) - Loop_QSSequencePrevious[{Stopped}].Grid_QSSP[catx].QSSP_2
			ActualMonth_Stop = (YearDiff_Stop * 12) + MonthDiff_Stop
			'debug.MsgBox(ActualMonth_Stop)
			
			if ActualMonth_Stop > ActualMonth_Start then
				Loop_QSSequencePrevious.Errors.Clear()
		    	Loop_QSSequencePrevious.Errors.AddNew("ERR33",Makestring(catx.Label,": ",IOM.Questions["hCustomMsgs"].Categories.ERR33.Label))
		    	Val_QSSequencePrevious = false
		    	exit function
			end if
			
			MRK_QSSP_X[catx].Started.Response = clong(makestring(Loop_QSSequencePrevious[{Started}].Grid_QSSP[catx].QSSP_2,"",Loop_QSSequencePrevious[{Started}].Grid_QSSP[catx].QSSP_1.Categories[Loop_QSSequencePrevious[{Started}].Grid_QSSP[catx].QSSP_1].Properties["month"]))
			MRK_QSSP_X[catx].Stopped.Response = clong(makestring(Loop_QSSequencePrevious[{Stopped}].Grid_QSSP[catx].QSSP_2,"",Loop_QSSequencePrevious[{Stopped}].Grid_QSSP[catx].QSSP_1.Categories[Loop_QSSequencePrevious[{Stopped}].Grid_QSSP[catx].QSSP_1].Properties["month"]))
			
	   		if (MRK_QSBIOLOGICDURATIONx <> null) then
				if (MRK_QSSP_X[catx].Started >= MRK_QSBIOLOGICDURATIONx) then
					Loop_QSSequencePrevious.Errors.Clear()
			    	Loop_QSSequencePrevious.Errors.AddNew("ERR34",Makestring(catx.Label,": ",IOM.Questions["hCustomMsgs"].Categories.ERR34.Label))
			    	Val_QSSequencePrevious = false
			    	exit function
				end if
				
				if (MRK_QSSP_X[catx].Stopped >= MRK_QSBIOLOGICDURATIONx) then
					Loop_QSSequencePrevious.Errors.Clear()
			    	Loop_QSSequencePrevious.Errors.AddNew("ERR35",Makestring(catx.Label,": ",IOM.Questions["hCustomMsgs"].Categories.ERR35.Label))
			    	Val_QSSequencePrevious = false
			    	exit function
				end if
			end if
		Next
   	Next
   	
   	'Punch the Answers before checking
   	For each cat in MRK_QSSequencePrevious.Categories
		MRK_QSSequencePrevious[cat].Started.Response = clong(makestring(Loop_QSSequencePrevious[{Started}].Grid_QSSP[cat].QSSP_2,"",Loop_QSSequencePrevious[{Started}].Grid_QSSP[cat].QSSP_1.Categories[Loop_QSSequencePrevious[{Started}].Grid_QSSP[cat].QSSP_1].Properties["month"]))
		MRK_QSSequencePrevious[cat].Stopped.Response = clong(makestring(Loop_QSSequencePrevious[{Stopped}].Grid_QSSP[cat].QSSP_2,"",Loop_QSSequencePrevious[{Stopped}].Grid_QSSP[cat].QSSP_1.Categories[Loop_QSSequencePrevious[{Stopped}].Grid_QSSP[cat].QSSP_1].Properties["month"]))
	Next
   	
   	Dim DateError
   	DateError = False
   	
   	for each cat in MRK_QSSequencePrevious.Categories
   		for each catx in MRK_QSSequencePrevious.Categories
   			if cat <> catx then
   				if (MRK_QSSequencePrevious[catx].Started >= MRK_QSSequencePrevious[cat].Started) AND (MRK_QSSequencePrevious[catx].Started =< MRK_QSSequencePrevious[cat].Stopped) then DateError = True

   				If  DateError = True Then

			    	Loop_QSSequencePrevious.Errors.AddNew("ERR36",Makestring(cat.Label,": ",IOM.Questions["hCustomMsgs"].Categories.ERR36.Label))
			    	Val_QSSequencePrevious = false

   			end if
   		next
